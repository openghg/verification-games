    def _step_summary(values):
        values = np.asarray(values)
        if values.size < 2:
            return {"step_min": np.nan, "step_median": np.nan, "step_max": np.nan}
        diffs = np.diff(values.astype("float64"))
        return {
            "step_min": float(np.nanmin(diffs)),
            "step_median": float(np.nanmedian(diffs)),
            "step_max": float(np.nanmax(diffs)),
        }

    def _coord_report(dim):

        fp_coord = fp_debug[dim]

        fp_values = np.asarray(fp_coord.values)

            "dim": dim,
            "flux_size": flux_values.size,
            "fp_size": fp_values.size,
            "flux_dtype": str(flux_values.dtype),
            "fp_dtype": str(fp_values.dtype),

            "fp_first": float(fp_values[0]),

            "same_size": bool(flux_values.shape == fp_values.shape),
        }

        row.update({f"fp_{key}": value for key, value in _step_summary(fp_values).items()})

            diff = flux_values.astype("float64") - fp_values.astype("float64")

        "fp_count": len(fp_time),

        "max_h_back_hours": max_lag,
        "expected_flux_start": expected_flux_time[0],
        "expected_flux_end": expected_flux_time[-1],
        "expected_flux_count": len(expected_flux_time),
        "missing_expected_flux_times": len(missing_flux_times),
        "fp_hourly": bool((pd.Series(fp_time).diff().dropna() == pd.Timedelta(hours=1)).all()),
        "flux_hourly": bool((pd.Series(flux_time).diff().dropna() == pd.Timedelta(hours=1)).all()),
    }
    print(pd.Series(time_report).to_string())

    if "staged_flux_for_run" in globals():
        try:

                "lon": 391,
            }
        )
        staged_flux_for_run = client.persist(staged_flux_for_run)
        wait(staged_flux_for_run)

    completed = []
    for site, start_date, end_date in iter_site_month_runs(sites=AVAILABLE_SHARED_STORE_SITES):
