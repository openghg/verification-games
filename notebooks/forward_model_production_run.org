def read_manifest_status(path: Path) -> dict[str, str | None]:
    if not path.exists():
        return {
            "manifest_status": None,
            "error_type": None,
            "error_message": None,
        }
    try:
        data = json.loads(path.read_text())
    except Exception as exc:
        return {
            "manifest_status": "unreadable_manifest",
            "error_type": type(exc).__name__,
            "error_message": str(exc),
        }
    return {
        "manifest_status": data.get("status"),
        "error_type": data.get("error_type"),
        "error_message": data.get("error_message"),
    }

** Optional bounded-concurrency writes
:PROPERTIES:
:CUSTOM_ID: optional-bounded-concurrency-writes
:END:
Use this only after the serial loop is understood. Do not run
=client.submit(run_site_month_safe)=: that puts a task on a worker which then
tries to drive Dask again from inside the worker. The pattern below builds each
lazy site/month write on the notebook client and submits only the resulting
Zarr write graph, capped by =MAX_IN_FLIGHT=.
It is disabled by default; validate it on one missing site/month before using
it for the full 2021 pass.

#+begin_src jupyter-python
RUN_BOUNDED_CONCURRENCY = False
MAX_IN_FLIGHT = 2


def _base_run_record(
    *,
    site: str,
    start_date,
    end_date,
    output_path: Path,
    status: str,
) -> ForwardModelRun:
    return ForwardModelRun(
        site=site.upper(),
        start_date=str(pd.Timestamp(start_date).date()),
        end_date=str(pd.Timestamp(end_date).date()),
        output_path=str(output_path),
        flux_zarr_path=str(STAGED_FLUX_ZARR),
        footprint_store="shared_store_zarr",
        footprint_domain="europe",
        footprint_species="co2",
        inlet=None,
        time_chunk=TIME_CHUNK,
        source_chunk=SOURCE_CHUNK,
        fillna_zero=False,
        created_at=datetime.now(UTC).isoformat(),
        status=status,
    )


def _failed_run_record(*, site: str, start_date, end_date, exc: Exception) -> ForwardModelRun:
    target = fp_dot_flux_output_path(
        FP_DOT_FLUX_OUTPUT_DIR,
        site=site.upper(),
        start_date=start_date,
    )
    return replace(
        _base_run_record(
            site=site,
            start_date=start_date,
            end_date=end_date,
            output_path=target,
            status="failed",
        ),
        error_type=type(exc).__name__,
        error_message=str(exc),
        error_traceback=traceback.format_exc(),
    )


def build_site_month_lazy_write(
    *,
    site: str,
    start_date,
    end_date,
    staged_flux=None,
    skip_existing: bool = True,
    overwrite: bool = False,
):
    site_upper = site.upper()
    target = fp_dot_flux_output_path(
        FP_DOT_FLUX_OUTPUT_DIR,
        site=site_upper,
        start_date=start_date,
    )
    if target.exists() and skip_existing and not overwrite:
        run = _base_run_record(
            site=site_upper,
            start_date=start_date,
            end_date=end_date,
            output_path=target,
            status="skipped_existing",
        )
        return run, None, None

    fp = get_month_footprint(
        site=site_upper,
        start_date=start_date,
        end_date=end_date,
        domain="europe",
        store="shared_store_zarr",
        species="co2",
    )
    flux = staged_flux if staged_flux is not None else open_staged_flux(STAGED_FLUX_ZARR)
    flux = select_flux_sources(flux)
    result = compute_fp_dot_flux(
        fp,
        flux,
        time_chunk=TIME_CHUNK,
        source_chunk=SOURCE_CHUNK,
        fillna_zero=False,
    )
    result = result.assign_attrs(
        {
            "site": site_upper,
            "start_date": str(pd.Timestamp(start_date).date()),
            "end_date": str(pd.Timestamp(end_date).date()),
            "footprint_store": "shared_store_zarr",
            "footprint_domain": "europe",
            "footprint_species": "co2",
            "inlet": "",
        }
    )

    target.parent.mkdir(parents=True, exist_ok=True)
    tmp = target.parent / f".{target.name}.{uuid4().hex}.tmp"
    ds = _zarr_safe_string_coords(result.to_dataset())
    delayed_write = ds.to_zarr(
        tmp,
        mode="w",
        consolidated=True,
        encoding=zarr_encoding(ds),
        compute=False,
    )
    run = _base_run_record(
        site=site_upper,
        start_date=start_date,
        end_date=end_date,
        output_path=target,
        status="written",
    )
    return run, delayed_write, tmp


if RUN_BOUNDED_CONCURRENCY:
    pending = list(iter_site_month_runs(sites=RUN_SITES, start=RUN_START, end=RUN_END))
    in_flight = {}
    bounded_completed = []

    staged_flux_for_bounded = None
    if PERSIST_STAGED_FLUX:
        staged_flux_for_bounded = open_staged_flux(STAGED_FLUX_ZARR).chunk(
            {"time": TIME_CHUNK, "source": SOURCE_CHUNK, "lat": 293, "lon": 391}
        )
        staged_flux_for_bounded = client.persist(staged_flux_for_bounded)
        wait(staged_flux_for_bounded)

    while pending or in_flight:
        while pending and len(in_flight) < MAX_IN_FLIGHT:
            site, start_date, end_date = pending.pop(0)
            try:
                run, delayed_write, tmp = build_site_month_lazy_write(
                    site=site,
                    start_date=start_date,
                    end_date=end_date,
                    staged_flux=staged_flux_for_bounded,
                    skip_existing=SKIP_EXISTING,
                    overwrite=OVERWRITE,
                )
            except Exception as exc:
                failed = _failed_run_record(
                    site=site,
                    start_date=start_date,
                    end_date=end_date,
                    exc=exc,
                )
                write_manifest(failed)
                bounded_completed.append(failed)
                continue

            if delayed_write is None:
                bounded_completed.append(run)
                continue

            future = client.compute(delayed_write)
            in_flight[future] = (run, tmp)
            print("Submitted write:", run.site, run.start_date, "in flight:", len(in_flight))

        if not in_flight:
            continue

        done, _ = wait(list(in_flight), return_when="FIRST_COMPLETED")
        for future in done:
            run, tmp = in_flight.pop(future)
            target = Path(run.output_path)
            try:
                future.result()
                if target.exists():
                    if OVERWRITE:
                        shutil.rmtree(target)
                    else:
                        raise FileExistsError(f"Output appeared during write: {target}")
                shutil.move(str(tmp), str(target))
                write_manifest(run)
                bounded_completed.append(run)
                print("Completed write:", run.site, run.start_date)
            except Exception as exc:
                if tmp is not None and Path(tmp).exists():
                    shutil.rmtree(tmp)
                failed = replace(
                    run,
                    status="failed",
                    error_type=type(exc).__name__,
                    error_message=str(exc),
                    error_traceback=traceback.format_exc(),
                )
                write_manifest(failed)
                bounded_completed.append(failed)
                print("Failed write:", failed.site, failed.start_date, failed.error_message)

    bounded_df = pd.DataFrame([run.__dict__ for run in bounded_completed])
    print(bounded_df["status"].value_counts(dropna=False).to_string())
